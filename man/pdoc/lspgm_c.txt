     
   C) GENERAL COMPUTATION PROGRAMS
   -------------------------------
     
   accumul  - PERFORMS ACCUMULATIONS                                   1/2 1
   addf     - ADDS UP TO 78 FILES (MINIMUM 2)                           78 1
   addlev   - ADDS/INTERPOLATEs LEVELS IN AN ARRAY                       1 1 C
   barx     - FAIT LA MOYENNE SELON X                                    2 1 C
   bary     - FAIT LA MOYENNE SELON y                                    2 1 C
   calmap   - CALCUL LES FACTEURS MAPSCALE ET DE CORIOLIS                1 2 C
   calpres  - CALCUL DE LOG(PI) SUR LES SURFACES DES NIVEAUX GAL-CHEN    4 3 C
   cdiv     - COMPLEX DIVISION OF TWO FILES (X/0=0)                      2 1
   cmlt     - COMPLEX MULTIPLICATION OF TWO FILES                        2 1
   complex  - COMPLEX ARRAYS TO/FROM REAL AND IMAGINARY PARTS          1/2 2/1
   conjmlt  - COMPLEX CONJUGATE MULTIPLICATION OF TWO COMPLEX FILES      2 1
   corpar   - COMPUTE CORIOLIS PARAMETER OF SAME TYPE/SIZE AS INPUT      1 1
   cosf     - COMPUTE THE COSINUS VALUE OF A FILE (IN RADIANS)           1 1 C
   delx     - FAIT LA DERIVEE selon X                                    2 1 C
   dely     - FAIT LA DERIVEE selon Y                                    2 1 C
   dertim   - TIME DIFFERENCE BETWEEN SUCCESIVE TIME STEPS IN A FILE     1 1
   diftim   - TIME DIFFERENCE OF TWO SETS IN THE SAME FILE               1 1 C
   divf     - DIVIDES TWO REAL FILES  (X/0=0)                            2 1
   dxdp     - VERTICAL PRESSURE DERIVATIVE OF A FILE                     1 1
   expone   - EXPONENTIATES A REAL FILE                                  1 1
   extrac   - ZEROES OUT ALL OF A GRID EXCEPT A WINDOW DEFINED IN INPUT  1 1 C
   fpow     - RAISES A REAL FILE TO A GIVEN POWER                        1 1 C
   flat     - FLATTEN THE LATERAL BOUNDARY IN THE NORMAL DIRECTION       1 1 C
   gadd     - ADDS TO EACH SET IN A FILE THE FIRST SET OF ANOTHER        1 1
   fstmdiag - CALCULATES MEANS, VARIANCES OF AN UNSTRUCTURED (RPN) FILE  1 2
   gdiv     - DIVIDES EACH SET IN A FILE BY THE FIRST SET OF ANOTHER     1 1
   globavg  - CALCULATES SURFACE AREA MEANS OF A SET OF FIELDS           1 1
   gmlt     - MULTIPLIES EACH SET IN A FILE BY THE FIRST SET OF ANOTHER  1 1
   grset    - CREATES A GRID DATA FIELD FOR TESTING PURPOSES             0 1 C G
   inttim   - TIME SUMMATION BETWEEN SUCCESIVE TIME STEPS IN A FILE      1 1
   kef      - COMPUTES KINETIC ENERGY  (U**2+V**2)/2.                    2 1
   loge     - NATURAL LOG OF A REAL FILE                                 1 1
   miroir   - BUILD GLOBAL ARRAYS FROM HEMISPHERIC ONES                  1 1 C
   mksymm   - BUILD SYMMETRIC GLOBAL ARRAYS FROM HEMISPHERIC ONES        1 1
   mkunsy   - BUILD ANTI-SYMMETRIC GLOBAL ARRAYS FROM HEMISPHERIC ONES   1 1
   mltf     - MULTIPLIES TWO REAL FILES                                  2 1
   msfact   - OUTPUT THE SURFACE MAP-SCALE-FACTORS USED BY globavg       1 1
   modulus  - MODULUS OF A FILE                                          1 1
   plagalc  - INTERPOLATES GAL-CHAN LEVEL GRIDS FROM P LEVELS FOR RCM    3 1 C
   pgen     - CREATES A FILE OF SAME TYPE AS INPUT WITH PRESSURE VALUES  1 1
   psdiv    - DIVIDES A FILE OF REAL SETS BY A ONE-LEVEL REAL FILE       2 1
   psmlt    - MULTIPLIES A FILE OF REAL SETS BY A ONE-LEVEL REAL FILE    2 1
   rgopr    - MULTIPLIES A MULTI0LEVEL FILE BY RGAS/PRESSURE             1 1
   rmax     - FINDS THE MAXIMUM FOR EACH RECORD IN A FILE                1 1
   rmin     - FINDS THE MINIMUM FOR EACH RECORD IN A FILE                1 1
   sinf     - COMPUTE THE SINUS VALUE OF A FILE (IN RADIANS)             1 1 C
   slice    - SELECTS ZONAL OR MERIDIONAL SECTIONS FROM FILE OF SETS     1 1
   smspec   - SPECTRAL FILTERING USING DISCREET SIN AND COS TRANSFORMS   1 2 C
   spconj   - COMPLEX CONJUGATE OF A COMPLEX FILE                        1 1
   sqroot   - TAKES THE SQUARE ROOT OF AN ARRAY (NO TYPE CHECK)          1 1
   square   - SQUARES GRID FIELDS                                        1 1
   stdev    - COMPUTES STANDARD DEVIATIONS                               1 1
   subf     - SUBTRACTS TWO FILES                                        2 1
   timavg   - COMPUTES MULTI-LEVEL TIME AVERAGE OF A FILE                1 1
   timcov   - COMPUTES COVARIANCES                                       2 1
   timdev   - COMPUTES TIME DEVIATIONS OF A FILE                         2 1
   timmax   - FINDS MAXIMUM TIME-SERIES VALUE AT EACH POINT              1 1
   timmin   - FINDS MINIMUM TIME-SERIES VALUE AT EACH POINT              1 1
   timsum   - (SAME AS PROGRAM INTTIM)                                   1 1
   varspec  - COMPUTE SPECTRUM FROM GRIDED VALUES                        1 3 C
   veint    - VERTICAL ETA INTEGRAL OF A FILE (FULL OR HALF LEVELS)      1 1 C
   vertinc  - VERTICAL PRESSURE INCREMENTS (DELTA P) LEVEL BY LEVEL      1 1
   vminmax  - FIND VERICAL MINIMUM OR MAXIMUMS IN A FILE                 1 1 C
   vpint    - VERTICAL PRESSURE INTEGRAL OF A FILE                       1 1
   vpintl   - VERTICAL PRESSURE INTEGRAL OF A FILE LEVEL BY LEVEL        1 1
   vpintv   - VERTICAL PRESSURE INTEGRAL WITH VARIABLE BOUNDARIES        2 1 C
   vsint    - VERTICAL SIGMA INTEGRAL OF A FILE (FULL OR HALF LEVELS)    1 1 C
   vsinth   - VERTICAL SIGMA INTEGRAL FOR HYBRID COORDINATE LEVELS       1 1 C
   ytox     - COMPUTES Z = Y**X                                          2 1
   xlin     - LINEAR OPERATION ON ONE FILE     Y=A*X+B                   1 1 C
   xpand    - MODIFIE LES DIMENSIONS(X,Y) D'UN TABLEAU 4D(X,Y,Z,T)       1 1 C
   xylin    - LINEAR OPERATION ON TWO FILES    Z=A*X+B*Y+C               2 1 C
